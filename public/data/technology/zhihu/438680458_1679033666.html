class Foo:
            
    def __init__(self, initial_balance=0):
        self.balance = initial_balance

    def deposit(self, amount):
        '''Deposit amount'''
        self.balance += amount

    def withdraw(self, amount):
        '''Withdraw amount'''
        self.balance -= amount

    def overdrawn(self):
        return self.balance < 0
